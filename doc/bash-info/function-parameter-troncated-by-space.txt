Solution:

function print_line() {
    [..]
    # data_method:
    case $(get_profile ".data$IDX.data_method") in
        sensors)
            DATA_STR=$(get_cache_sensor "$(get_profile ".data$IDX.sensors_namespace")")   <-- here "$(get_profile ".data$IDX.sensors_namespace"
        ;;
        get_mem)
            DATA_STR=$(get_mem)
        ;;
    esac
